13def sort_multi_array(data: list, descending: bool = False) -> list:
14 """
15 Sorts a multi-dimensional array by the second value in each array using lambda shorthand, returns a copy of the list
16 instead of sorting the original incase caller wants to keep the original list. If the descending parameter is set
17 then the data will be sorted in reverse order. The second array in the list must be a list of numbers.
18
19 @param data: The list to be sorted
20 @param descending: Weather to sort in descending order or not. By default, this is False.
21 @return: The sorted list (a copy of the original list)
22 """
23
24 # Unpack the data into two lists, one for the names and one for the scores
25 names, scores = data[0], data[1]
26
27 # Sort the data by the scores
28 sorted_data = sorted(zip(names, scores), key=lambda x: x[1], reverse=descending)
29
30 # Put the data back into the original format
31 result = list(zip(*sorted_data))
32 result = [list(r) for r in result]
33
34 return result

37def string_bool(text: str) -> bool:
38 """
39 A replacement for the built-in bool function that allows for the conversion of a string with the text
40 "True/False" to a bool. It will raise a ValueError if the text is not "True" or "False" otherwise it will return
41 the value.
42
43 @note string_bool is a replacement for the built-in bool function as when using bool() to convert a string to a
44 bool it will return True if there is any text in the string and False if there isn't, however I need it to return
45 True if the string is "True" and False if the string is "False"
46
47 @note: Do not use this when loading from JSON as the JSON module will convert the string to a bool
48
49 @param text: The text to convert to a bool
50 @return: The bool value of the text. Value Error if the text is not "True" or "False"
51 """
52 if text == "True":
53 return True
54 elif text == "False":
55 return False
56 else:
57 raise ValueError()

60def ip_address(text: str) -> Exception or str:
61 """
62 A type to use to check if the string is a valid IP address, if it is then the string will be returned,
63 if it isn't than a ValueError will be raised. Must adhere to the IP address format of V4.
64
65 @param text: The text to check
66 @return: True if the text is a valid IP address, False if it isn't
67 """
68
69 # Check if the text is a valid IP address
70 if text.count(".") != 3:
71 raise ValueError()
72
73 # Check if the octets are valid
74 for part in text.split("."):
75 if not part.isdigit():
76 raise ValueError()
77
78 if not 0 <= int(part) <= 255:
79 raise ValueError()
80
81 return text

175def try_convert(variable: object, type_to_convert: type, supress_errors: bool = False) -> object:
176 """
177 Try to convert a variable to a type, if it fails then return None
178
179 @param variable: The variable to convert
180 @param type_to_convert: The type to convert the variable to
181 @param supress_errors: Weather to supress errors or not (By default: False)
182 @return: The converted variable or None if it failed
183 """
184 if variable is None:
185 return None
186
187 try:
188 return type_to_convert(variable)
189 except ValueError:
190 if not supress_errors:
191 error("Invalid input")
