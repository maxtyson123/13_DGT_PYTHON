Quiz Game 0.54
A Python programming language based application designed to train players for a quiz. The program uses the OpenTDB API to generate questions for the players to practice their quiz skills. The program uses the Requests module to send HTTP requests to the trivia website and then utilizes a self-built renderer and game module to generate a quiz with a simple user interface. After the game, players can track their progression of skills through stats such as time spent. The program allows for the addition of bots to play alongside the player and can be configured to set the accuracy of their AI. The program is flexible and can expand menu sizes and is written to handle unexpected user input and corrupted JSON files.

417def text_in_divider(item_to_print: str, wrap: WrapMode = WrapMode.TRUNCATE) -> None:
418 """
419 Prints the item in between two divider symbols and wraps the text if it is too long by default using Truncate but
420 will use whatever is passed in the wrap parameter.
421
422 @param item_to_print: The text to print
423 @param wrap: How the text should be wrapped if it is too long (Truncate by default)
424 @return: None
425 """
426 text_length = len(Colour.clean_text(item_to_print))
427 console_space_free = console_width - (divider_symbol_size * 2)
428
429 # If the text is longer than the console width then wrap it
430 if text_length > console_space_free:
431 match wrap:
432 case WrapMode.CHAR:
433 # Print the current text in the divider, cutting off the text at the console width
434 print(divider_symbol + item_to_print[:console_space_free] + divider_symbol)
435
436 # Store the text to print as adding a space for readability
437 item_to_print = " " + item_to_print[console_space_free:]
438
439 # Re-run the function as many times as needed and then return so the original, un wrapped, text isn't
440 # printed
441 text_in_divider(item_to_print, wrap)
442 return
443
444 case WrapMode.WORD:
445 # Split the text into words
446 words = item_to_print.split(" ")
447
448 # Store the current size
449 current_size = 0
450
451 item_to_print = ""
452
453 # Loop through all the words
454 for word in words:
455 # If this word is bigger then the entire console then split it up
456 if len(word) > console_space_free:
457 # Join the words together
458 item_to_print = " ".join(words)
459
460 # Print the item in the divider, split at console width
461 print(divider_symbol + item_to_print[:console_space_free] + divider_symbol)
462 current_size += console_space_free
463
464 # Store the text to print as adding a space for readability. Cut off the text already written
465 # and ignore the padding space
466 item_to_print = " " + item_to_print[current_size:]
467
468 # Re-run the function as many times as needed and then return so the original, un wrapped,
469 # text isn't printed
470 text_in_divider(item_to_print, wrap)
471 return
472
473 # If the current size plus the word length is less than the console width
474 if current_size + len(word) < console_space_free:
475 # Add the word to the current size
476 current_size += len(word)
477
478 # Add a space to the current size
479 current_size += 1
480
481 # Add the word to the text to print
482 item_to_print += word + " "
483
484 else:
485 # When here is reached then it is time to break out of the loop as the next word cant fit in
486 # the space left for this line
487
488 # Add spacing so the whole line is used
489 item_to_print += " " * (console_space_free - current_size)
490
491 # Break out of the loop
492 break
493
494 # Print the item in the divider
495 print(divider_symbol + item_to_print + divider_symbol)
496
497 # Join the words together
498 item_to_print = " ".join(words)
499
500 # Store the text to print as adding a space for readability. Cut off the text already written and
501 # ignore the padding space
502 item_to_print = " " + item_to_print[current_size:]
503
504 # Re-run the function as many times as needed and then return so the original, un wrapped, text isn't
505 # printed
506 text_in_divider(item_to_print, wrap)
507 return
508
509 case WrapMode.TRUNCATE:
510 # Slice the text at console width - 2 (for the border)
511 item_to_print = item_to_print[:console_space_free]
512
513 case WrapMode.NONE:
514 pass
515
516 # The length of the text, minus console width, minus 2 for the border
517 width_left = console_space_free - text_length
518 print(divider_symbol + item_to_print + " " * width_left + divider_symbol)
