28def generate_new_save_file():
29 """
30 Generates a new save file name by searching through the data folder for a name that is not already taken in the
31 format of "Game_0.json" (where 0 is the number of the save file). If more than 1000 save files are found, then an
32 exception is raised to prevent infinite looping as most likely something has gone wrong.
33
34 @return: The name of the save file
35 """
36 # Get all the save files
37 already_saved = get_saved_games()
38
39 # Create a name for the save file
40 name = "Game_"
41 name_index = 0
42 save_name = name + str(name_index) + ".json"
43
44 # Check if the save file already exists
45 while save_name in already_saved:
46 name_index += 1
47 save_name = name + str(name_index) + ".json"
48
49 # If there is more than 1000 save files, then something probably went wrong
50 if name_index > 1000:
51 raise Exception("Could not find a save file name")
52
53 return save_name

229 # zero error)
