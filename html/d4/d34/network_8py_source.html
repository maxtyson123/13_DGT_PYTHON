36 def to_bytes(self) -> bytes:
37 """
38 Creates a JSON object from the attributes of this object and encodes it in utf-8

44 "recipient": self.recipient,

101 def accept_connection(self, sock: socket) -> None:
102 """
103 Accept a connection from a client, saves the client and registers it with the selector. The data (sometimes
104 called key_data) is a namespace with the address of the client and the data to send and receive.
105
106 @param sock: The socket to accept the connection from
107 """
108 # Accept the connection
109 connection, address = sock.accept()
110
111 # Print the address
112 debug_message(f"Accepted connection from {address}", "network_server")
113
114 # Don't block the process as we want to be able to accept multiple connections
115 connection.setblocking(False)
116
117 # Create the data object
118 data = types.SimpleNamespace(socket_adress=address, recieved_bytes=b"", send_bytes=b"")
119
120 # Create the events. They are now read and write so set those bits
121 events = selectors.EVENT_READ | selectors.EVENT_WRITE
122
123 # Register the connection with the selector
124 self.selector.register(connection, events, data=data)
125
126 # Add the connection to the list of clients
127 self.clients.append(connection)
128 self.client_names.append(address)

137 debug_message(f"Closing connection on {sock}", "network_server")
138
139 # Unregister the socket from the selector
140 self.selector.unregister(sock)
141
142 # Close the socket
143 sock.close()
144
145 # Remove the socket from the list of clients
146 for sock_index in range(len(self.clients)):
147 if self.clients[sock_index] == sock:
148 # Remove the client and name
149 self.clients.pop(sock_index)
150 self.client_names.pop(sock_index)
151 break

153 def service_connection(self, key: object, mask: object) -> None:
154 """
155 Service a connection from a client. This is called when the client has data to send or is ready to receive
156 data. Data is read in 4096 byte chunks so messages shouldnt be sent directly one after another or will cause
157 issues on the recving end. Data is then sent to the handle_data_received function. If there is data to send
158 then it is sent to the handle_data_send function.
159
160 @param key: The key to the client
161 @param mask: The mask to the client
162 """
163
164 try:
165 sock = key.fileobj
166 data = key.data
167
168 # If the socket has its read bit set
169 if mask & selectors.EVENT_READ:
170
171 # Get the data
172 recv_data = sock.recv(4096)
173
174 # If there is no data then the connection has been closed
175 if recv_data:
176 self.handle_data_received(sock, data, recv_data)
177
178 else:
179 # Print the address
180 debug_message(f"Closing connection on {data.socket_adress}", "network_server")
181
182 # Close the connection
183 self.close_connection(sock)
184
185 # If the socket has its write bit set
186 if mask & selectors.EVENT_WRITE:
187
188 # If there is data to send
189 if data.send_bytes:
190 self.handle_data_send(sock, data, data.send_bytes)
191
192 except Exception as e:
193 self.handle_error(sock, data, e)

289 def run(self) -> None:
290 """
291 Run the client, listening for data from the server refreshing every second. If there is data then it is read
292 at 4096 bytes per chunk and then checked to see if it is a valid json string. If it is then it is sent to the
293 handle_data_received function. If there is no data then the connection has been closed and the client is closed.
294 """
295 try:
296
297 while True:
298
299 events = self.selector.select(timeout=1)
300
301 for key, mask in events:
302
303 try:
304 sock = key.fileobj
305 data = key.data
306
307 # If the socket has its read bit set
308 if mask & selectors.EVENT_READ:
309
310 # Store all the data parts in a list
311 fragments = []
312 recv_data = b''
313
314 # While there is data to read
315 while True:
316 data = sock.recv(4096)
317 fragments.append(data)
318 recv_data = b''.join(fragments)
319
320 try:
321 decoded = recv_data.decode()
322 json.loads(decoded)
323 break
324 except ValueError:
325 # TODO: This is not the best way of doing this as a malicious server could send
326 # broken JSON and cause the client to loop forever
327 continue
328
329 # If there is no data then the connection has been closed
330 if recv_data:
331 self.handle_data_received(sock, data, recv_data)
332
333 # If the socket has its write bit set
334 if mask & selectors.EVENT_WRITE:
335
336 # If there is data to send
337 if data.send_bytes:
338 self.handle_request(sock, data, data.send_bytes)
339
340 except Exception as e:
341 self.handle_error(sock, data, e)
342
343 # OSErrors thrown here are caused when the socket is being deleted, so can ignore them
344 except OSError:
345 pass

412 def handle_data_received(self, sock: socket, key_data: object, recv_data: bytes) -> None:
413 """
414 Handle data received from a client. This function can handle the client_join message type and the sync_player
415 message. Both of these actions involve updating the game's users array with the new user data if the request
416 was handled without error. Upon handling the messages the server will send a error response to the socket if
417 the game is full or has already started or if the username is already taken. If the game has already started
418 then the server will only allow the client to join if they have a name that is already in the game's users
419 but not connected, which is treated as a reconnect.
420
421 @param sock: The socket to handle the data from
422 @param key_data: The data from the selector key
423 @param recv_data: The data received from the client (same as key_data.receive_bytes but kept for convenience)
424 """
425
426 # Convert the data to a message
427 message = QuizMessage(None, None, None, None)
428 message.from_bytes(recv_data)
429
430 # Handle the message
431 match message.message_type:
432 case "client_join":
433 # Check if the game has started
434 if self.game.game_started:
435 self.send_message(sock, "Game has already started", "server_error")
436 return
437
438 # Check if the game is full
439 if len(self.game.users) >= self.game.max_players:
440 self.send_message(sock, "Game is full", "server_error")
441 return
442
443 # Convert the user to an object
444 self.game.users.append(message.message)
445 temp_index = len(self.game.users) - 1
446 self.game.convert_users()
447
448 is_new_player = True
449
450 # Check if the username is taken
451 for user_index in range(len(self.game.users) - 1):
452
453 # If the username is taken
454 user = self.game.users[user_index]
455 if user.name == self.game.users[temp_index].name:
456 # If the user is taken and connected then the name is taken so send an error
457 if user.is_connected:
458 self.send_message(sock, "Username is taken", "server_error")
459 # Remove the user from the list
460 self.game.users.pop(temp_index)
461 return
462
463 debug_message(f"Removing index {temp_index}", "network_server")
464
465 # Remove the user from the list as will use the old one
466 self.game.users.pop(temp_index)
467 temp_index = user_index
468
469 # This means the server is continuing a game so send the user their progress
470 is_new_player = False
471 self.sync_game()
472 time.sleep(0.5)
473
474 break
475
476 # If this server is continuing a game then don't allow new players to join
477 if is_new_player and self.game.game_loaded:
478 self.send_message(sock,
479 "Server is continuing game, must rejoin using same name. New players arent allowed",
480 "server_error")
481 # Remove the user from the list
482 self.game.users.pop(temp_index)
483 return
484
485 self.game.convert_users()
486
487 # Set the sockets name to the user's name
488 self.client_names[self.clients.index(sock)] = self.game.users[temp_index].name
489
490 # User is now connected
491 self.game.users[temp_index].is_connected = True
492 debug_message(f"Player {message.message['name']} has joined the game", "network_server")
493
494 case "sync_player":
495 user = message.message
496
497 index = 0
498
499 # Find the user to update
500 for user_index in range(len(self.game.users)):
501 if self.game.users[user_index].name == user["name"]:
502 self.game.users[user_index] = user
503 self.game.users[user_index]["is_connected"] = True
504 index = user_index
505 break
506
507 # Convert the users to objects
508 self.game.convert_users()
509 debug_message(f"Player: {self.game.users[index].name} has synced", "network_server")
510
511 case _:
512 debug_message(f"Unhandled message: {message.message}", "network_server")

746def api_get_questions(amount: int, category: int, difficulty: str, question_type: str) -> list:
747 """
748 Gets questions from the API at https://opentdb.com/api.php and returns them as a list of dictionaries
749
750 @param amount: How many questions to get (Max 50)
751 @param category: The category of the questions (index, offset by 9)
752 @param difficulty: The difficulty of the questions
753 @param question_type: The type of the questions
754 @return: A list of dictionaries containing the questions
755 """
756
757 # Get the setup data
758 install_package("requests")
759
760 import requests
761
762 redo = True
763 api_fix = 0
764
765 user_data = UserData()
766
767 while redo:
768
769 # Create the URL
770 url = f"https://opentdb.com/api.php?amount={amount}"
771
772 # Ignore the options if they are "Any" (or none because 'convert_question_settings_to_api' already does this)
773 # since the API gives any by default
774
775 if question_type is not None and api_fix < 1:
776 url += f"&type={question_type}"
777
778 if difficulty != "Any" and api_fix < 2:
779 url += f"&difficulty={difficulty.lower()}"
780
781 if category is not None and api_fix < 3:
782 url += f"&category={category}"
783
784 debug_message(url, "API")
785
786 # Get the questions from the API
787 response = requests.get(url)
788
789 # Check if the was any errors
790 if response.status_code != 200:
791 error("Failed to get questions from the API")
792 return []
793
794 # Debug
795 debug_message(str(response.json()), "API")
796
797 # Check for errors
798 match response.json()["response_code"]:
799 case 0:
800 pass # No errors, just good to have a defined case so that I don't forget it
801 case 1:
802 error("No results found")
803 if user_data.auto_fix_api:
804 print("Auto fixing API error...")
805 api_fix += 1
806 continue
807
808 case 2:
809 error("Invalid parameter")
810
811 # Return the questions
812 return response.json()["results"]
